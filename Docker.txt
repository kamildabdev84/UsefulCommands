-----------------------------------------------------------------
1.  Show available images
-----------------------------------------------------------------

    docker images;

-----------------------------------------------------------------
2.  Show running containers
-----------------------------------------------------------------
	
    docker ps;

-----------------------------------------------------------------
3.  Remove image
-----------------------------------------------------------------
	
    docker rmi 7f9d92fc996f

-----------------------------------------------------------------
4.  Remove container
-----------------------------------------------------------------
	
    docker rm  7f9d92fc996f

-----------------------------------------------------------------
5.  Save and compress image
-----------------------------------------------------------------

    docker save example-1.0 > example-1.0.tar; 
    gzip example-1.0.tar > example-1.0.tar.gz

-----------------------------------------------------------------
6.  Load image from tar.gz
-----------------------------------------------------------------
	
    docker load example-1.0.tar.gz	

-----------------------------------------------------------------
7.  Run container in background
-----------------------------------------------------------------
	
    docker run -d example-1.0

-----------------------------------------------------------------
8.  Run container with exposed port 8080
-----------------------------------------------------------------

    docker run -d -p 8080:8080 example-1.0

-----------------------------------------------------------------
9.  Run bash in container
-----------------------------------------------------------------

    docker exec -it <mycontainer> bash

-----------------------------------------------------------------
10. Remove all containers
-----------------------------------------------------------------

    docker rm $(sudo docker ps -aq);

-----------------------------------------------------------------
11. Stop all containers
-----------------------------------------------------------------

    docker stop $(docker ps -aq)

-----------------------------------------------------------------
12. Remove all images
-----------------------------------------------------------------
    
    docker rmi  $(docker images -aq);

-----------------------------------------------------------------
13. Run docker-compose script interactively
-----------------------------------------------------------------
    
    docker-compose up

-----------------------------------------------------------------
14. Run docker-compose script as a deamon
-----------------------------------------------------------------

    docker-compose up -d

-----------------------------------------------------------------
15. Run these commands when user cannot run docker without sudo
-----------------------------------------------------------------

    sudo usermod -a -G docker $USER
    reboot
